CREATE TABLE questionnaire(id INT NOT NULL auto_increment, title VARCHAR (255) NOT NULL, description VARCHAR(255) NOT NULL, intro VARCHAR (255) NOT NULL, enabled BOOLEAN, PRIMARY KEY (id));CREATE TABLE question (id INT NOT NULL auto_increment, theme VARCHAR (255), title VARCHAR (255), text VARCHAR (255), image LONGBLOB, PRIMARY KEY (id));
CREATE TABLE questionlist(id INT NOT NULL auto_increment, airid INT NOT NULL, qid INT NOT NULL, enabled BOOLEAN, PRIMARY KEY(id), FOREIGN KEY (airid) REFERENCES questionnaire (id), FOREIGN KEY (qid) REFERENCES question (id));
CREATE TABLE user(id INT NOT NULL auto_increment, name VARCHAR (25) NOT NULL, lastname VARCHAR(25) NOT NULL, email VARCHAR(50), username VARCHAR(20) NOT NULL, password VARCHAR (20) NOT NULL, admin BOOLEAN NOT NULL, enabled BOOLEAN NOT NULL,profession VARCHAR(25), denied BOOLEAN NOT NULL, PRIMARY KEY (id));
CREATE TABLE form (id INT NOT NULL auto_increment, airid INT NOT NULL, userid INT NOT NULL, memo VARCHAR (255), category VARCHAR (50), relation VARCHAR (50), completed BOOLEAN, checkedreport BOOLEAN, PRIMARY KEY (id), FOREIGN KEY (airid) REFERENCES questionnaire (id), FOREIGN KEY (userid) REFERENCES user (id)); 
CREATE TABLE client (id INT NOT NULL auto_increment,start BOOLEAN NOT NULL, formid INT NOT NULL, age INT, function VARCHAR (50), hash VARCHAR (20), done BOOLEAN, PRIMARY KEY (id), FOREIGN KEY (formid) REFERENCES form (id));CREATE TABLE answer(id INT NOT NULL auto_increment, qid INT NOT NULL, clientid INT NOT NULL, score INT, help BOOLEAN, final BOOLEAN, PRIMARY KEY (id), FOREIGN KEY (qid) REFERENCES question (id), FOREIGN KEY (clientid) REFERENCES client (id));
